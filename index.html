iPreview jQuery Image Preview Plugin

Simple Example

$('#preview-file').ipreview();

Resize Example

$('#preview-file2').ipreview({
	resize: true,
	maxwidth: 100,
	maxheight: 200
});

Advance Example

$('#preview-file3').ipreview({
	success: function(ele, data){
		ele.attr('src', data);
		alert('Image Replace!');
	}
});
